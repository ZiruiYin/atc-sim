# Core simulator (Flask backend + local play). The GitHub Pages deployment
# runs the sim in-browser via Pyodide and needs NONE of these.

waitress>=3.0   # production WSGI server (main.py prefers it over the dev server)

# --- AUTO planner (backend-only) ---------------------------------------------

# (torch is a large download; skip this section if you don't need AUTO.)

# Needed for the /tts route + tts/ package. Not used by the Pyodide build.

# faster-whisper: controller audio -> raw text (stt/ package, /stt route).
# transformers:   raw text -> sim command DSL (dsl_parser/, reuses the torch above).

faster-whisper>=1.0
transformers>=4.40
